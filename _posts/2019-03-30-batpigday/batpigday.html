sim_n <- function(k = 3,
                   min_n = 20,
                   max_n = 200,
                   prop = 0.3,
                   wide = FALSE) {
  assert_that(min_n > 0,
              msg = "minimum sample size must be positive")
  assert_that(max_n > 0,
              msg = "maximum sample size must be positive")
  assert_that(min_n <= max_n,
              msg = "minimum sample size cannot exceed maximum sample size")
